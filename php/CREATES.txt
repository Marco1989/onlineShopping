CREATE TABLE Region (RID INTEGER AUTO_INCREMENT, RegionName VARCHAR(10), RegionManager VARCHAR(50), PRIMARY KEY (RID));

CREATE TABLE Store (SID INTEGER AUTO_INCREMENT, Manager VARCHAR(50), RID INTEGER, Street VARCHAR(30), City VARCHAR(20), State VARCHAR(2), ZIP VARCHAR(5), PRIMARY KEY (SID), FOREIGN KEY (RID) REFERENCES Region (RID));

CREATE TABLE SalesEmployees (EID INTEGER AUTO_INCREMENT, Name VARCHAR(50), SID INTEGER, Street VARCHAR(30), City VARCHAR(20), State VARCHAR(2), ZIP VARCHAR(5), Email VARCHAR(50), Title VARCHAR(20), Salary FLOAT, Login VARCHAR(20), Password VARCHAR(30), Role INTEGER, PRIMARY KEY (EID), FOREIGN KEY (SID) REFERENCES Store (SID));

CREATE TABLE Product (PID INTEGER AUTO_INCREMENT, Pname VARCHAR(30), Price FLOAT, Category ENUM ('Tablet', 'Laptop', 'Desktop', 'Media Device', 'Phone', 'Camera'), Status CHAR, Description VARCHAR(50), PRIMARY KEY (PID));

CREATE TABLE ProductStore (PID INTEGER , SID INTEGER, Quantity INTEGER, PRIMARY KEY (PID, SID));

CREATE TABLE Customer (CID INTEGER AUTO_INCREMENT, Name VARCHAR(50) NOT NULL, Email VARCHAR(50), Street VARCHAR(30), City VARCHAR(20), State VARCHAR(2), ZIP VARCHAR(5), PRIMARY KEY (CID));

CREATE TABLE HomeCustomer (CID INTEGER , Gender VARCHAR(1), Age INTEGER, MaritalStatus VARCHAR(1), Income FLOAT, PRIMARY KEY (CID), FOREIGN KEY (CID) REFERENCES Customer (CID));

CREATE TABLE BusinessCustomer (CID INTEGER , CompanyName VARCHAR(20), Category VARCHAR(20), GrossIncome FLOAT, AcctNumber VARCHAR(10), PRIMARY KEY (CID), FOREIGN KEY (CID) REFERENCES Customer (CID));

CREATE TABLE Transactions (TID INTEGER AUTO_INCREMENT, PID INTEGER, CID INTEGER, EID INTEGER, TDate DATE, Quantity INTEGER, PRIMARY KEY (TID, PID), FOREIGN KEY (PID) REFERENCES Product (PID), FOREIGN KEY (CID) REFERENCES Customer (CID), FOREIGN KEY (EID) REFERENCES SalesEmployees (EID));

CREATE TABLE PriceRecord (TID INTEGER, PID INTEGER, HPrice FLOAT, PRIMARY KEY (TID, PID), FOREIGN KEY (TID) REFERENCES Transactions (TID), FOREIGN KEY (PID) REFERENCES Product (PID));
